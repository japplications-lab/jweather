const SPC_BASE  = 'https://www.spc.noaa.gov';
const NWS_BASE  = 'https://api.weather.gov';
const NOAA_BASE = 'https://mapservices.weather.noaa.gov';
const IEM_BASE  = 'https://mesonet.agron.iastate.edu';

const ALLOWED_PREFIXES = [
  '/products/outlook/day1otlk',
  '/products/outlook/day2otlk',
  '/products/outlook/day3otlk',
  '/products/outlook/day1probotlk',
  '/products/outlook/day2probotlk',
  '/products/outlook/imgs_v2/',
];

const CORS_HEADERS = {
  'Access-Control-Allow-Origin': '*',
  'Access-Control-Allow-Methods': 'GET, HEAD, POST, OPTIONS',
  'Access-Control-Allow-Headers': '*',
};

const NWS_TTL = {
  '/points/':             3600,
  '/forecast':             300,
  '/forecast/hourly':      300,
  '/alerts/active':         30, // Lower TTL for faster alert updates
  '/stations':            3600,
  '/observations/latest':  120,
};

function nwsTtl(path) {
  for (const [prefix, ttl] of Object.entries(NWS_TTL)) {
    if (path.includes(prefix)) return ttl;
  }
  return 300;
}

const SPC_CAT = {
  'TSTM': { color: '#c8eec8', label: 'General Storms'  },
  'MRGL': { color: '#6abf6a', label: 'Low Risk'         },
  'SLGT': { color: '#f0e040', label: 'Some Risk'        },
  'ENH':  { color: '#e88020', label: 'Elevated Risk'    },
  'MDT':  { color: '#e04040', label: 'High Risk'        },
  'HIGH': { color: '#ee30a0', label: 'Extreme Risk'     },
};
const DN_TO_CAT = { '2': 'TSTM', '3': 'MRGL', '4': 'SLGT', '5': 'ENH', '6': 'MDT', '8': 'HIGH' };

function pointInRing(pt, ring) {
  let inside = false;
  const [x, y] = pt;
  for (let i = 0, j = ring.length - 1; i < ring.length; j = i++) {
    const [xi, yi] = ring[i], [xj, yj] = ring[j];
    if (((yi > y) !== (yj > y)) && (x < (xj - xi) * (y - yi) / (yj - yi) + xi)) inside = !inside;
  }
  return inside;
}

function pointInFeature(pt, feature) {
  const geom = feature.geometry;
  if (!geom) return false;
  const polys = geom.type === 'Polygon'
    ? [geom.coordinates]
    : geom.type === 'MultiPolygon' ? geom.coordinates : [];
  for (const poly of polys) {
    if (pointInRing(pt, poly[0])) return true;
  }
  return false;
}

async function fetchSpcOutlookForPoint(lat, lng) {
  const days = [
    { url: 'day1otlk.json', label: 'Today'     },
    { url: 'day2otlk.json', label: 'Tomorrow'  },
    { url: 'day3otlk.json', label: 'in 2 Days' },
  ];
  const order = ['HIGH', 'MDT', 'ENH', 'SLGT', 'MRGL', 'TSTM'];
  const pt = [lng, lat];
  const results = [];

  await Promise.all(days.map(async ({ url, label }) => {
    try {
      const res = await fetch(`${SPC_BASE}/products/outlook/${url}`, {
        headers: {
          'User-Agent': 'Mozilla/5.0 (compatible; jWeather/1.0)',
          'Referer': 'https://www.spc.noaa.gov/',
        },
        cf: { cacheTtl: 300 },
      });
      if (!res.ok) return;
      const gj = await res.json();
      if (!gj.features?.length) return;

      let best = null, bestOrder = -1;
      for (const feat of gj.features) {
        const raw = (feat.properties?.LABEL || feat.properties?.label || feat.properties?.DN || '')
          .toString().toUpperCase().trim();
        const key = SPC_CAT[raw] ? raw : (DN_TO_CAT[raw] || null);
        if (!key) continue;
        const rank = order.indexOf(key);
        if (rank > bestOrder && pointInFeature(pt, feat)) { best = key; bestOrder = rank; }
      }
      if (best) results.push({ risk: best, dayLabel: label, ...SPC_CAT[best] });
    } catch (e) {
      console.warn(`SPC day check failed for ${url}:`, e);
    }
  }));

  return results.length ? results : null;
}

export default {
  async fetch(request, env) {

    if (request.method === 'OPTIONS') {
      return new Response(null, { status: 204, headers: CORS_HEADERS });
    }

    const url = new URL(request.url);

    if (url.pathname.startsWith('/iem/')) {
      const iemPath = url.pathname.replace(/^\/iem/, '') + url.search;
      const upstream = `${IEM_BASE}${iemPath}`;

      try {
        const res = await fetch(upstream, {
          headers: {
            'User-Agent': 'Mozilla/5.0 (compatible; jWeather/1.0)',
            'Referer':     'https://mesonet.agron.iastate.edu/',
          },
          cf: { cacheTtl: 60 },
        });

        const body = await res.arrayBuffer();
        const ct   = res.headers.get('content-type') || 'image/png';

        return new Response(body, {
          status: res.status,
          headers: {
            ...CORS_HEADERS,
            'Content-Type':  ct,
            'Cache-Control': 'public, max-age=60',
          },
        });
      } catch (err) {
        return new Response(JSON.stringify({ error: err.message }), {
          status: 502,
          headers: { ...CORS_HEADERS, 'Content-Type': 'application/json' },
        });
      }
    }

    if (url.pathname.startsWith('/nws/')) {
      const nwsPath = url.pathname.replace(/^\/nws/, '') + url.search;
      const upstream = `${NWS_BASE}${nwsPath}`;
      const ttl = nwsTtl(nwsPath);

      try {
        const res = await fetch(upstream, {
          headers: {
            'User-Agent': 'jWeather/1.0 (contact@yourdomain.com)',
            'Accept': 'application/geo+json',
          },
          cf: { cacheTtl: ttl },
        });

        const body = await res.arrayBuffer();
        const ct = res.headers.get('content-type') || 'application/geo+json';

        return new Response(body, {
          status: res.status,
          headers: {
            ...CORS_HEADERS,
            'Content-Type': ct,
            'Cache-Control': `public, max-age=${ttl}`,
          },
        });
      } catch (err) {
        return new Response(JSON.stringify({ error: err.message }), {
          status: 502,
          headers: { ...CORS_HEADERS, 'Content-Type': 'application/json' },
        });
      }
    }

    if (url.pathname.startsWith('/noaa/')) {
      const noaaPath = url.pathname.replace(/^\/noaa/, '') + url.search;
      const upstream = `${NOAA_BASE}${noaaPath}`;

      try {
        const res = await fetch(upstream, {
          headers: {
            'User-Agent': 'jWeather/1.0',
            'Accept': 'application/json, application/geo+json',
          },
          cf: { cacheTtl: 300 },
        });

        const body = await res.arrayBuffer();
        const ct = res.headers.get('content-type') || 'application/json';

        return new Response(body, {
          status: res.status,
          headers: {
            ...CORS_HEADERS,
            'Content-Type': ct,
            'Cache-Control': 'public, max-age=300',
          },
        });
      } catch (err) {
        return new Response(JSON.stringify({ error: err.message }), {
          status: 502,
          headers: { ...CORS_HEADERS, 'Content-Type': 'application/json' },
        });
      }
    }

    if (url.pathname === '/outlook') {
      const lat = parseFloat(url.searchParams.get('lat'));
      const lng = parseFloat(url.searchParams.get('lng'));

      if (isNaN(lat) || isNaN(lng)) {
        return new Response(JSON.stringify({ error: 'Missing or invalid lat/lng parameters' }), {
          status: 400,
          headers: { ...CORS_HEADERS, 'Content-Type': 'application/json' },
        });
      }

      const result = await fetchSpcOutlookForPoint(lat, lng);

      return new Response(JSON.stringify(result ?? null), {
        status: 200,
        headers: {
          ...CORS_HEADERS,
          'Content-Type': 'application/json',
          'Cache-Control': 'public, max-age=300',
        },
      });
    }

    if (url.pathname === '/ai') {
      if (request.method !== 'POST') {
        return new Response('Method not allowed', { status: 405, headers: CORS_HEADERS });
      }

      const apiKey = env.SPC;

      if (!apiKey) {
        return new Response(JSON.stringify({ error: 'GROQ_API_KEY is not set' }), {
          status: 500,
          headers: { ...CORS_HEADERS, 'Content-Type': 'application/json' },
        });
      }

      let body;
      try { body = await request.json(); }
      catch (e) {
        return new Response(JSON.stringify({ error: 'Invalid JSON: ' + e.message }), {
          status: 400,
          headers: { ...CORS_HEADERS, 'Content-Type': 'application/json' },
        });
      }

      const messages = [];
      if (body.system) messages.push({ role: 'system', content: body.system });
      if (body.messages) messages.push(...body.messages);

      const groqPayload = {
        model: body.model || 'llama-3.3-70b-versatile',
        max_tokens: body.max_tokens || 1000,
        messages,
      };

      let groqRes;
      try {
        groqRes = await fetch('https://api.groq.com/openai/v1/chat/completions', {
          method: 'POST',
          headers: {
            'Content-Type': 'application/json',
            'Authorization': `Bearer ${apiKey}`,
          },
          body: JSON.stringify(groqPayload),
        });
      } catch (err) {
        return new Response(JSON.stringify({ error: 'Fetch to Groq failed: ' + err.message }), {
          status: 502,
          headers: { ...CORS_HEADERS, 'Content-Type': 'application/json' },
        });
      }

      const rawText = await groqRes.text();

      if (!groqRes.ok) {
        return new Response(JSON.stringify({ error: `Groq ${groqRes.status}: ${rawText}` }), {
          status: 500,
          headers: { ...CORS_HEADERS, 'Content-Type': 'application/json' },
        });
      }

      let data;
      try { data = JSON.parse(rawText); }
      catch (e) {
        return new Response(JSON.stringify({ error: 'Groq response not valid JSON' }), {
          status: 500,
          headers: { ...CORS_HEADERS, 'Content-Type': 'application/json' },
        });
      }

      const text = data.choices?.[0]?.message?.content || '';
      return new Response(JSON.stringify({ content: [{ type: 'text', text }] }), {
        status: 200,
        headers: { ...CORS_HEADERS, 'Content-Type': 'application/json' },
      });
    }

    const spcPath = url.pathname.replace(/^\/proxy/, '') + url.search;

    const allowed = ALLOWED_PREFIXES.some(p => spcPath.startsWith(p));
    if (!allowed) {
      return new Response('Not allowed', { status: 403, headers: CORS_HEADERS });
    }

    const targetUrl = `${SPC_BASE}${spcPath}`;

    try {
      const response = await fetch(targetUrl, {
        headers: {
          'User-Agent': 'Mozilla/5.0 (compatible; jWeather/1.0)',
          'Referer': 'https://www.spc.noaa.gov/',
        },
        cf: { cacheTtl: 300 },
      });

      const contentType = response.headers.get('content-type') || 'application/octet-stream';

      return new Response(response.body, {
        status: response.status,
        headers: {
          ...CORS_HEADERS,
          'Content-Type': contentType,
          'Cache-Control': 'public, max-age=300',
        },
      });
    } catch (err) {
      return new Response(`Proxy error: ${err.message}`, {
        status: 502,
        headers: CORS_HEADERS,
      });
    }
  },
};
