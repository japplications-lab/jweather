UPDATED MAPPING LOGIC (Add this inside your existing script tag) function updateWarningsMap(data) { if (!data || !data.features) return; warningsLayer.clearLayers(); data.features.forEach(feature => { const event = feature.properties.event; const desc = (feature.properties.description || "").toLowerCase(); let color = "#ffeb3b"; // Default yellow // Logic for extreme warnings based on NWS standards if (event === "Tornado Warning") { color = "#ff0000"; // Red if (desc.includes("particularly dangerous situation")) color = "#ff1493"; // PDS Pink if (desc.includes("tornado emergency")) color = "#ff00ff"; // Emergency Magenta } else if (event === "Severe Thunderstorm Warning") { color = "#ffa500"; // Orange if (desc.includes("destructive")) color = "#e65100"; // Deeper orange for destructive tags } else if (event === "Flash Flood Warning") { color = "#8b0000"; // Dark Red if (desc.includes("flash flood emergency")) color = "#ff00ff"; } else if (event.includes("Watch")) { color = event.includes("Tornado") ? "#ffe600" : "#db7093"; } if (feature.geometry) { L.geoJSON(feature, { style: { color: color, weight: 3, fillOpacity: 0.35, fillColor: color } }).addTo(warningsLayer); } }); } // 2. LIVE DATA FETCHING (Ensures most recent API data) async function fetchLiveSevereData() { try { // Fetching Extreme/Severe alerts from NWS API const response = await fetch('https://api.weather.gov/alerts/active?status=actual&message_type=alert&severity=Extreme,Severe'); const data = await response.json(); updateWarningsMap(data); } catch (error) { console.error("Error fetching live severe data:", error); } } // 3. INITIALIZATION & REFRESH // Run immediately on load fetchLiveSevereData(); // Automatically refresh data every 2 minutes (120,000 ms) setInterval(fetchLiveSevereData, 120000);
